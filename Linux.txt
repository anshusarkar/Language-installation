for C lang installation use command

sudo apt install gcc 

for C++ installation use command

sudo apt install g++

for Python installtion use command

sudo apt install python

for java installation use command 

sudo apt install default-jdk # This will install the defualt-jdk for the specific system of linux

for R installation use command 

sudo apt install r-base r-base-dev # for R-lang

sudo apt-get install gdebi-core
wget https://download2.rstudio.org/server/bionic/amd64/rstudio-server-2021.09.1-372-amd64.deb
sudo gdebi rstudio-server-2021.09.1-372-amd64.deb # for rstudio 

or download the R stdio deb package for debian then use 

sudo dpkg -i "filename.deb"
